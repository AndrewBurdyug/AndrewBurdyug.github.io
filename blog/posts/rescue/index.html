Chroot manjaro

# mount /dev/nvmen0p1 /mnt
## mount /dev/nvmen0p1 /mnt/boot/efi  # optional - in case you are running in UEFI mode, again, replace with the correct device name
# mount -t proc proc /mnt/proc
# mount --rbind /dev /mnt/dev
# mount --rbind /sys /mnt/sys
# mount --rbind /run /mnt/run
# mount --rbind /etc/resolv.conf /mnt/etc/resolv.conf
# chroot /mnt /bin/bash
